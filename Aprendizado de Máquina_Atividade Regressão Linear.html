# Agrupando os dados por continente
dados.rename(columns={
    'World regions according to OWID': 'regiao'
}, inplace=True)
regioes = dados['regiao'].unique()

# Criando gr√°ficos separados por continuente
for regiao in regioes:
    dados_regiao = dados[dados['regiao'] == regiao]
    
    X = dados_regiao[['renda_per_capita']]  # Vari√°vel independente
    y = dados_regiao['felicidade']          # Vari√°vel dependente
    
    modelo = LinearRegression() # modelo regress√£o
    modelo.fit(X, y)
    y_pred = modelo.predict(X)
    coeficiente = modelo.coef_[0]
    intercepto = modelo.intercept_
    r2 = r2_score(y, y_pred)
    print(f"R¬≤ para {regiao}: {r2:.3f}")
    print(f"Coeficiente: {coeficiente}")
    print(f"Intercepto: {intercepto}")
  
    plt.figure(figsize=(10, 6)) #plotagem 
    sns.scatterplot(x='renda_per_capita', y='felicidade', data=dados_regiao, color='cyan', label='Pais')
    for i in range(len(dados_regiao)):
        plt.text(dados_regiao['renda_per_capita'].iloc[i], dados_regiao['felicidade'].iloc[i], 
                 dados_regiao['pais'].iloc[i], fontsize=8, alpha=0.7)
    plt.plot(dados_regiao['renda_per_capita'], y_pred, color='red', label='Regress√£o Linear')
    plt.title(f'Regress√£o Linear - {regiao}: Felicidade vs Renda per Capita')
    plt.xlabel('Renda per Capita)')
    plt.ylabel('Felicidade (Cantril ladder score)')
    plt.legend()
    plt.show()
